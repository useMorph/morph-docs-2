### Example

```python
from morph_lib.ai.openai.code_generation import text_to_markdown
import os

@morph.func
def func_name(context):
    prompt = context.vars["prompt"]
    fig = text_to_markdown(
		prompt,
		os.environ["OPENAI_API_KEY"],
		"gpt-4o",
	)
	code = fig.code
	print(code)

	return fig.content
```
